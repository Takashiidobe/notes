class Codec:
    def serialize(self, root):
        res = []
        
        def dfs(node):
            if not node:
                res.append("N")
                return
            res.append(str(node.val))
            dfs(node.left)
            dfs(node.right)
            
        dfs(root)    
        ans = ",".join(res)
        return ans
            

    def deserialize(self, data):
        self.i = 0
        data = data.split(',')

            if data[self.i] == "N":
                self.i += 1
                return None
